// alias 쓰는 건 좋은데 잘 모르는 상태에서 사용하면 원래 명령어 잊을 수도 있으니까 
// 많이 익숙해진다음 적용하도록 하자

[alias]
	a = add .
	b = branch
	df = diff
	co = checkout
	s = status -s
	st = stash push -m
	stl = stash list
	stp = stash pop
	sta = stash apply
	l = log
	hs = log --graph --all --pretty=format:'%C(yellow)[%ad]%C(reset) %C(green)[%h]%C(reset) | %C(white)%s %Cgreen(%cr) %C(bold blue){{%an}}%C(reset) %C(blue)%d%C(reset)' --date=short
	sm = switch main
	sw = switch
	rs = restore .

	alias vim="nvim"
	alias vi="nvim"
	alias gm="gitmoji -c"
	alias la="lsd -al --no-symlink"
	alias ls="lsd --no-symlink"
	alias ll="lsd -l --no-symlink"
	alias lt="lsd --tree --no-symlink"
	alias tl="tmux ls"
	alias td="tmux detach"
	alias ta="tmux a"
