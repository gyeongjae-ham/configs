""" mark 표시하기
let g:mark_a_state = 0
function! ToggleMarkA()
    if g:mark_a_state == 0
        normal! mA
        let g:mark_a_state = 1
    else
        delmarks A
        let g:mark_a_state = 0
    endif
endfunction

nnoremap qq :call ToggleMarkA()<CR>
nnoremap zz `A
